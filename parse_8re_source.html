00342 {
00343   char *buf;
00344   int len, oldlen, nmemb;
00345 
00346   if (scanner->token > scanner->limit) {
00347     /* this shouldn't happen */
00348     return 0;

00350   if (scanner->f && !feof(scanner->f)) {
00351     oldlen = scanner->limit - scanner->token;
