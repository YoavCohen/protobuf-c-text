00344 {
00345   char *buf;

00347 
00348   if (scanner->token > scanner->limit) {
00349     /* this shouldn't happen */
00350     return 0;

00352   if (scanner->f && !feof(scanner->f)) {
00353     oldlen = scanner->limit - scanner->token;

00355     buf = allocator->alloc(allocator->allocator_data, len);

00619 {
00620   switch (t->id) {
00621     case TOK_BAREWORD:
00622       state->field = protobuf_c_message_descriptor_get_field_by_name(
00623           state->msgs[state->current_msg]->descriptor, t->bareword);
00624       if (state->field) {
00625         if (state->field->label != PROTOBUF_C_LABEL_REQUIRED

00628           return state_error(state, t,

00637       }

00639     case TOK_CBRACE:
00640       if (state->current_msg > 0) {

00646       break;
00647     case TOK_EOF:

00651       }

00653       break;
00654     default:

00691         if (state->field->label == PROTOBUF_C_LABEL_OPTIONAL

00708               STRUCT_MEMBER(ProtobufCMessage *, msg, state->field->offset),
