00341 static int
00342 fill(Scanner *scanner, ProtobufCAllocator *allocator)
00343 {
00344   char *buf;
00345   int len, oldlen, nmemb;
00346 
00347   if (scanner->token > scanner->limit) {
00348     /* this shouldn't happen */
00349     return 0;
00350   }
00351   if (scanner->f && !feof(scanner->f)) {
00352     oldlen = scanner->limit - scanner->token;
00353     len = CHUNK + oldlen;
00354     buf = allocator->alloc(allocator->allocator_data, len);
00355     if (!buf) {
00356       return -1;
00357     }
00358     memcpy(buf, scanner->token, oldlen);
00359     nmemb = fread(buf + oldlen, 1, CHUNK, scanner->f);
00360     if (nmemb != CHUNK) {
00361       /* Short read.  eof.  Append nul. */
00362       len = oldlen + nmemb;
00363       buf[len] = '\0';
00364     }
00365     /* Reset the world to use buf. */
00366     scanner->cursor = &buf[scanner->cursor - scanner->token];
00367     scanner->limit = buf + len;
00368     scanner->token = buf;
00369     allocator->free(allocator->allocator_data, scanner->buffer);
00370     scanner->buffer = buf;
00371     scanner->marker = buf;
00372   }
00373 
00374   return scanner->limit >= scanner->cursor? 1: 0;

00611 static StateId
00612 state_open(State *state, Token *t)
00613 {
00614   switch (t->id) {
00615     case TOK_BAREWORD:
00616       state->field = protobuf_c_message_descriptor_get_field_by_name(
00617           state->msgs[state->current_msg]->descriptor, t->bareword);
00618       if (state->field) {
00619         if (state->field->label != PROTOBUF_C_LABEL_REQUIRED
00620             && state->field->label != PROTOBUF_C_LABEL_OPTIONAL
00621             && state->field->label != PROTOBUF_C_LABEL_REPEATED) {
00622           return state_error(state, t,
00623               "Internal error: unknown label type %d for '%s'.",
00624               state->field->label, state->field->name);
00625         }
00626         return STATE_ASSIGNMENT;
00627       } else {
00628         return state_error(state, t, "Can't find field '%s' in message '%s'.",
00629                            t->bareword,
00630                            state->msgs[state->current_msg]->descriptor->name);
00631       }
00632       break;
00633     case TOK_CBRACE:
00634       if (state->current_msg > 0) {
00635         state->current_msg--;
00636       } else {
00637         return state_error(state, t, "Extra closing brace found.");
00638       }
00639       return STATE_OPEN;
00640       break;
00641     case TOK_EOF:
00642       if (state->current_msg > 0) {
00643         return state_error(state, t, "Missing '%d' closing braces.",
00644             state->current_msg);
00645       }
00646       return STATE_DONE;
00647       break;
00648     default:
00649       return state_error(state, t,
00650                          "Expected element name or '}'; found '%s' instead.",
00651                          token2txt(t));

00681         ProtobufCMessage **tmp;

00683 
00684         /* Don't assign over an existing message. */
00685         if (state->field->label == PROTOBUF_C_LABEL_OPTIONAL

00697         if (state->field->label == PROTOBUF_C_LABEL_REPEATED) {

00700                                     state->field->quantifier_offset);

00702               STRUCT_MEMBER(ProtobufCMessage *, msg, state->field->offset),

00711           tmp[n_members - 1] = NULL;
